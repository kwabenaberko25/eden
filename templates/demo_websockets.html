Code Example

from eden.websocket import WebSocketRouter

ws = WebSocketRouter(prefix="/ws", auth_required=True)

@ws.on_connect
async def on_connect(socket, manager):
    await manager.broadcast({"type": "joined"}, channel="chat")

@ws.on("message")
async def on_message(socket, data, manager):
    await manager.broadcast(data, channel="chat", exclude=socket)

@ws.on_disconnect
async def on_disconnect(socket, manager):
    await manager.broadcast({"type": "left"}, channel="chat")

app.routes.extend(ws.routes)
